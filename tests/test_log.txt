from backend.redaction.pii import redact_and_tokenize, de_tokenize
import backend.db.setup_db as setup_db
setup_db.setup_db()

text = 'Inspector John Smith can be reached at john.smith@example.com or 214-555-0199.'
clean = redact_and_tokenize(text, 'test-doc-1', setup_db.DB_PATH)
restored = de_tokenize(clean, 'test-doc-1', setup_db.DB_PATH)

with open('pii_text_output.txt', 'w') as f:
    f.write('INPUT:\n' + text + '\n\n')
    f.write('REDACTED:\n' + clean + '\n\n')
    f.write('RESTORED (de-tokenized):\n' + restored + '\n')

print('done')
"
